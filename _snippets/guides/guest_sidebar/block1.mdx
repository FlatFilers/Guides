import api from "@flatfile/api";

export default function flatfileEventListener(listener) {
  listener.on(
    "upload:completed",
    async ({ context: { spaceId, environmentId } }) => {
      try {
        const updateSpace = await api.spaces.update(spaceId, {
          environmentId,
          metadata: {
            sidebarConfig: {
              showSidebar: false,
            },
