```js javascript
red.use(
  recordHook("contacts", (record) => {
    const value = record.get("firstName");
    if (typeof value === "string") {
      record.set("firstName", value.toLowerCase());
    } else {
      record.addError("firstName", "Invalid first name");
    }
    return record;
  }),
);
```
